async function copy(text: string) {
  try {
    await navigator.clipboard.writeText(text);
    showToast("–°–∫–æ–ø–∏—Ä–æ–≤–∞–Ω–æ ‚ú®");
  } catch {
    // fallback

}

/** Tilt effect */
function onTilt(e: MouseEvent) {
  const card = e.currentTarget as HTMLElement
  const rect = card.getBoundingClientRect()
  const x = (e.clientX - rect.left) / rect.width
  const y = (e.clientY - rect.top) / rect.height

  const rotateY = (x - 0.5) * 10
  const rotateX = (0.5 - y) * 10

  card.style.setProperty('--rx', `${rotateX}deg`)
  card.style.setProperty('--ry', `${rotateY}deg`)
  card.style.setProperty('--glx', `${x * 100}%`)
  card.style.setProperty('--gly', `${y * 100}%`)
}

function resetTilt(e: MouseEvent) {
  const card = e.currentTarget as HTMLElement
  card.style.setProperty('--rx', `0deg`)
  card.style.setProperty('--ry', `0deg`)
}

function addToCart(item: MenuItem) {
  const found = cart.value.find((c) => c.id === item.id)
  if (found) found.qty += 1
  else cart.value.push({ ...item, qty: 1 })
}

function inc(line: CartLine) {
  const found = cart.value.find((c) => c.id === line.id)
  if (found) found.qty += 1
}
function dec(line: CartLine) {
  const found = cart.value.find((c) => c.id === line.id)
  if (!found) return
  found.qty -= 1
  if (found.qty <= 0) cart.value = cart.value.filter((c) => c.id !== line.id)
}

const cartCount = computed(() => cart.value.reduce((sum, c) => sum + c.qty, 0))

// ------- Background pointer parallax -------
function setRootVar(name: string, value: string) {
  document.documentElement.style.setProperty(name, value)
}
function onPointerMove(e: PointerEvent) {
  const w = window.innerWidth || 1
  const h = window.innerHeight || 1
  const x = (e.clientX / w - 0.5) * 2 // -1..1
  const y = (e.clientY / h - 0.5) * 2 // -1..1
  setRootVar("--mx", x.toFixed(4))
  setRootVar("--my", y.toFixed(4))
}
function onPointerLeave() {
  setRootVar("--mx", "0")
  setRootVar("--my", "0")
}

// ------- Tilt cards (pointer-based) -------
function tiltMove(e: PointerEvent) {
  const el = e.currentTarget as HTMLElement | null
  if (!el) return
  const r = el.getBoundingClientRect()
  const px = (e.clientX - r.left) / Math.max(1, r.width)
  const py = (e.clientY - r.top) / Math.max(1, r.height)
  el.style.setProperty("--px", String(px))
  el.style.setProperty("--py", String(py))
  el.style.setProperty("--tilt", "1")
}
function tiltLeave(e: PointerEvent) {
  const el = e.currentTarget as HTMLElement | null
  if (!el) return
  el.style.setProperty("--tilt", "0")
}

// ------- Reveal on scroll -------
onMounted(() => {
  // section observer
  const sectionIds = ["coffee", "drinks", "crepes"] as const
  const secEls = sectionIds.map((id) => document.getElementById(id)).filter(Boolean) as HTMLElement[]
  const secObs = new IntersectionObserver(
    (entries) => {
      const visible = entries.filter((x) => x.isIntersecting).sort((a, b) => (b.intersectionRatio ?? 0) - (a.intersectionRatio ?? 0))[0]
      if (!visible?.target?.id) return
      activeSection.value = visible.target.id as any
    },
    { root: null, threshold: [0.25, 0.4, 0.6] }
  )
  secEls.forEach((el) => secObs.observe(el))

  // reveal observer
  const revealEls = Array.from(document.querySelectorAll<HTMLElement>(".reveal"))
  const revObs = new IntersectionObserver(
    (entries) => {
      for (const it of entries) {
        if (it.isIntersecting) (it.target as HTMLElement).classList.add("is-visible")
      }
    },
    { threshold: 0.15 }
  )
  revealEls.forEach((el) => revObs.observe(el))
})

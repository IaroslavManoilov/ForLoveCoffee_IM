<script setup>
import { computed, onBeforeUnmount, onMounted, ref } from "vue";
import { useRouter } from "vue-router";
import infLogo from "@/assets/logo.svg";
import bg from "@/assets/im2.png";

const router = useRouter();
const goMenu = () => router.push("/menu");
const goBooking = () => router.push("/booking");

const rootEl = ref(null);

const reduceMotion = ref(false);
const mq = typeof window !== "undefined" ? window.matchMedia("(prefers-reduced-motion: reduce)") : null;
const setReduceMotion = () => (reduceMotion.value = Boolean(mq?.matches));

const clamp = (v, min, max) => Math.min(max, Math.max(min, v));

/* parallax + tilt */
const mx = ref(0);
const my = ref(0);
const tx = ref(0);
const ty = ref(0);
let raf = 0;

const tick = () => {
    tx.value += (mx.value - tx.value) * 0.08;
    ty.value += (my.value - ty.value) * 0.08;
    raf = requestAnimationFrame(tick);
};

onMounted(() => {
    setReduceMotion();
    mq?.addEventListener?.("change", setReduceMotion);
    raf = requestAnimationFrame(tick);
});

onBeforeUnmount(() => {
    cancelAnimationFrame(raf);
    mq?.removeEventListener?.("change", setReduceMotion);
});

const onMove = (e) => {
    if (reduceMotion.value) return;
    const r = e.currentTarget.getBoundingClientRect();
    mx.value = clamp(((e.clientX - r.left) / r.width - 0.5) * 2, -1, 1);
    my.value = clamp(((e.clientY - r.top) / r.height - 0.5) * 2, -1, 1);
};
const onLeave = () => {
    mx.value = 0;
    my.value = 0;
};

const leftStyle = computed(() => {
    if (reduceMotion.value) return {};
    return { transform: `translate3d(${tx.value * 6}px, ${ty.value * 4}px, 0)` };
});
const cardStyle = computed(() => {
    if (reduceMotion.value) return {};
    const rx = ty.value * -4;
    const ry = tx.value * 6;
    const dx = tx.value * -10;
    const dy = ty.value * -8;
    return { transform: `translate3d(${dx}px, ${dy}px, 0) rotateX(${rx}deg) rotateY(${ry}deg)` };
});

/* button magnet */
const onBtnMove = (e) => {
    if (reduceMotion.value) return;
    const el = e.currentTarget;
    const r = el.getBoundingClientRect();
    const nx = ((e.clientX - r.left) / r.width - 0.5) * 2;
    const ny = ((e.clientY - r.top) / r.height - 0.5) * 2;
    el.style.setProperty("--bx", `${clamp(nx, -1, 1) * 4}px`);
    el.style.setProperty("--by", `${clamp(ny, -1, 1) * 3}px`);
};
const onBtnLeave = (e) => {
    const el = e.currentTarget;
    el.style.setProperty("--bx", `0px`);
    el.style.setProperty("--by", `0px`);
};

/* Open/Closed logic (09:00-22:00) */
const OPEN_H = 9;
const CLOSE_H = 22;

const now = ref(new Date());
let timer = 0;

const isOpen = computed(() => {
    const h = now.value.getHours();
    const m = now.value.getMinutes();
    const minutes = h * 60 + m;
    return minutes >= OPEN_H * 60 && minutes < CLOSE_H * 60;
});
const statusText = computed(() => (isOpen.value ? "–û—Ç–∫—Ä—ã—Ç–æ —Å–µ–π—á–∞—Å" : "–°–µ–π—á–∞—Å –º—ã –∑–∞–∫—Ä—ã—Ç—ã"));

onMounted(() => {
    timer = window.setInterval(() => (now.value = new Date()), 30_000);
});
onBeforeUnmount(() => {
    if (timer) window.clearInterval(timer);
});

/* Hearts particles */
const particleStyle = (n) => {
    const seed = (n * 9301 + 49297) % 233280;
    const rnd = seed / 233280;

    const x = Math.round(rnd * 92 + 4);
    const y = Math.round(((rnd * 7) % 1) * 84 + 6);
    const s = (rnd * 0.75 + 0.35).toFixed(2);
    const d = Math.round(rnd * 2200 + 200);
    const a = (rnd * 0.28 + 0.10).toFixed(2);

    return { left: `${x}%`, top: `${y}%`, opacity: a, transform: `scale(${s})`, animationDelay: `${d}ms` };
};

<script setup lang="ts">
const threeCanvas = ref<HTMLCanvasElement | null>(null);

const logoUrl = "/logo.svg";
const clamp = (v: number, min: number, max: number) => Math.min(max, Math.max(min, v));

/* UI state */
const menuOpen = ref(false);

/* reduced motion */
const reduceMotion = ref(false);
let mq: MediaQueryList | null = null;
const setReduceMotion = () => (reduceMotion.value = Boolean(mq?.matches));

/* parallax */
const mx = ref(0);
const my = ref(0);
const tx = ref(0);
const ty = ref(0);
let raf = 0;

const tick = () => {
    tx.value += (mx.value - tx.value) * 0.08;
    ty.value += (my.value - ty.value) * 0.08;
    raf = requestAnimationFrame(tick);
};

const onMove = (e: MouseEvent) => {
    if (reduceMotion.value) return;
    const el = e.currentTarget as HTMLElement;
    const r = el.getBoundingClientRect();
    mx.value = clamp(((e.clientX - r.left) / r.width - 0.5) * 2, -1, 1);
    my.value = clamp(((e.clientY - r.top) / r.height - 0.5) * 2, -1, 1);
};
const onLeave = () => { mx.value = 0; my.value = 0; };

const introStyle = computed(() => {
    if (reduceMotion.value) return {};
    return { transform: `translate3d(${tx.value * 8}px, ${ty.value * 6}px, 0)` };
});
const cardStyle = computed(() => {
    if (reduceMotion.value) return {};
    const rx = ty.value * -6;
    const ry = tx.value * 8;
    const dx = tx.value * -14;
    const dy = ty.value * -10;
    return { transform: `translate3d(${dx}px, ${dy}px, 0) rotateX(${rx}deg) rotateY(${ry}deg)` };
});

/* magnetic buttons */
const onBtnMove = (e: MouseEvent) => {
    if (reduceMotion.value) return;
    const el = e.currentTarget as HTMLElement;
    const r = el.getBoundingClientRect();
    const nx = ((e.clientX - r.left) / r.width - 0.5) * 2;
    const ny = ((e.clientY - r.top) / r.height - 0.5) * 2;
    el.style.setProperty("--bx", `${clamp(nx, -1, 1) * 6}px`);
    el.style.setProperty("--by", `${clamp(ny, -1, 1) * 5}px`);
};
const onBtnLeave = (e: MouseEvent) => {
    const el = e.currentTarget as HTMLElement;
    el.style.setProperty("--bx", "0px");
    el.style.setProperty("--by", "0px");
};

/* open/closed */
const OPEN_H = 9;
const CLOSE_H = 22;
const now = ref(new Date());
let timer = 0;

const isOpen = computed(() => {
    const h = now.value.getHours();
    const m = now.value.getMinutes();
    const minutes = h * 60 + m;
    return minutes >= OPEN_H * 60 && minutes < CLOSE_H * 60;
});
const statusText = computed(() => (isOpen.value ? "–û—Ç–∫—Ä—ã—Ç–æ —Å–µ–π—á–∞—Å" : "–°–µ–π—á–∞—Å –∑–∞–∫—Ä—ã—Ç–æ"));

/* scroll progress */
const scrollY = ref(0);
const scrollProgress = ref(0);

const onScroll = () => {
    const y = window.scrollY || 0;
    scrollY.value = y;

    const doc = document.documentElement;
    const max = Math.max(1, doc.scrollHeight - doc.clientHeight);
    scrollProgress.value = clamp(y / max, 0, 1);
};

/* reveal */
const aboutEl = ref<HTMLElement | null>(null);
const menuEl = ref<HTMLElement | null>(null);
const reviewsEl = ref<HTMLElement | null>(null);
const visitEl = ref<HTMLElement | null>(null);

const inView = ref({ about: false, menu: false, reviews: false, visit: false });
let io: IntersectionObserver | null = null;

const observe = (elRef: Ref<HTMLElement | null>, key: keyof typeof inView.value) => {
    const el = elRef.value;
    if (!el || !io) return;
    el.dataset.revealKey = String(key);
    io.observe(el);
};

const scrollTo = (id: string) => {
    const el = document.getElementById(id);
    if (!el) return;
    el.scrollIntoView({ behavior: reduceMotion.value ? "auto" : "smooth", block: "start" });
};
const goSection = (id: string) => {
    menuOpen.value = false;
    requestAnimationFrame(() => scrollTo(id));
};

/* THREE background (dynamic import => no TS errors) */
let cleanupThree: null | (() => void) = null;

const initThree = async () => {
    if (!threeCanvas.value) return () => { };

    const THREE = await import("three");

    const canvas = threeCanvas.value;

    const renderer = new THREE.WebGLRenderer({
        canvas,
        antialias: true,
        alpha: true,
        powerPreference: "high-performance",
    });
    renderer.setPixelRatio(Math.min(window.devicePixelRatio || 1, 2));

    const scene = new THREE.Scene();
    scene.fog = new THREE.FogExp2(0x000000, 0.08);

    const camera = new THREE.PerspectiveCamera(55, 1, 0.1, 200);
    camera.position.set(0, 0, 10);

    scene.add(new THREE.AmbientLight(0xffffff, 0.6));
    const key = new THREE.DirectionalLight(0xffffff, 0.75);
    key.position.set(6, 4, 8);
    scene.add(key);

    const rim = new THREE.PointLight(0xffd1d1, 1.2, 40);
    rim.position.set(-6, -2, 10);
    scene.add(rim);

    const group = new THREE.Group();
    scene.add(group);

    // better materials
    const matA = new THREE.MeshPhysicalMaterial({
        color: 0xb24a4a,
        metalness: 0.35,
        roughness: 0.22,
        clearcoat: 0.8,
        clearcoatRoughness: 0.25,
        emissive: 0x1a0000,
        emissiveIntensity: 0.25,
    });
    const matB = new THREE.MeshStandardMaterial({
        color: 0xffffff,
        metalness: 0.12,
        roughness: 0.18,
        transparent: true,
        opacity: 0.18,
    });

    // centerpiece: torus knot + rings + ‚Äúbubbles‚Äù
    const knot = new THREE.Mesh(new THREE.TorusKnotGeometry(1.3, 0.28, 280, 20, 2, 5), matA);
    knot.position.set(0.25, 0.1, 0);
    group.add(knot);

    const ringGeo = new THREE.TorusGeometry(1.75, 0.08, 18, 160);
    const ring1 = new THREE.Mesh(ringGeo, matB);
    ring1.position.set(-0.4, -0.2, -1.2);
    ring1.rotation.set(0.9, 0.1, 0.3);
    group.add(ring1);

    const ring2 = new THREE.Mesh(ringGeo, matB);
    ring2.position.set(0.8, 0.5, -2.3);
    ring2.rotation.set(0.2, 0.8, 0.6);
    group.add(ring2);

    const orbGeo = new THREE.SphereGeometry(0.22, 22, 22);
    const orbs: THREE.Mesh[] = [];
    for (let i = 0; i < 26; i++) {
        const m = new THREE.Mesh(orbGeo, matB);
        m.position.set((Math.random() - 0.5) * 9, (Math.random() - 0.5) * 5.5, -Math.random() * 9);
        m.scale.setScalar(Math.random() * 1.8 + 0.55);
        group.add(m);
        orbs.push(m);
    }

    const clock = new THREE.Clock();

    const resize = () => {
        const w = window.innerWidth;
        const h = window.innerHeight;
        renderer.setSize(w, h, false);
        camera.aspect = w / h;
        camera.updateProjectionMatrix();
    };
    resize();
    window.addEventListener("resize", resize, { passive: true });

    let anim = 0;

    const animate = () => {
        anim = requestAnimationFrame(animate);
        const t = clock.getElapsedTime();
        const motion = reduceMotion.value ? 0.2 : 1;

        group.rotation.y = (t * 0.18 + tx.value * 0.6) * motion;
        group.rotation.x = (t * 0.12 + ty.value * 0.5) * motion;

        knot.rotation.z = t * 0.22 * motion;
        ring1.rotation.z = -t * 0.14 * motion;
        ring2.rotation.z = t * 0.12 * motion;

        for (let i = 0; i < orbs.length; i++) {
            const o = orbs[i];
            o.position.y += Math.sin(t * 0.6 + i) * 0.0010 * motion;
            o.position.x += Math.cos(t * 0.45 + i) * 0.0008 * motion;
        }

        camera.position.z = 10 + scrollProgress.value * 1.7;
        camera.position.y = -scrollProgress.value * 0.9;

        renderer.render(scene, camera);
    };

    animate();

    return () => {
        window.removeEventListener("resize", resize);
        cancelAnimationFrame(anim);

        scene.traverse((obj: any) => {
            if (!obj?.isMesh) return;
            obj.geometry?.dispose?.();
            if (Array.isArray(obj.material)) obj.material.forEach((m: any) => m.dispose?.());
            else obj.material?.dispose?.();
        });

        renderer.dispose?.();
    };
};

onMounted(async () => {
    mq = window.matchMedia("(prefers-reduced-motion: reduce)");
    setReduceMotion();
    mq.addEventListener?.("change", setReduceMotion);

    raf = requestAnimationFrame(tick);

    timer = window.setInterval(() => (now.value = new Date()), 30_000);

    onScroll();
    window.addEventListener("scroll", onScroll, { passive: true });

    io = new IntersectionObserver(
        (entries) => {
            for (const e of entries) {
                if (!e.isIntersecting) continue;
                const key = e.target.dataset.revealKey as keyof typeof inView.value | undefined;
                if (key) inView.value[key] = true;
            }
        },
        { threshold: 0.18, rootMargin: "0px 0px -10% 0px" }
    );

    observe(aboutEl, "about");
    observe(menuEl, "menu");
    observe(reviewsEl, "reviews");
    observe(visitEl, "visit");

    try {
        cleanupThree = await initThree();
    } catch (e) {
        console.warn("Three init failed:", e);
    }
});

onBeforeUnmount(() => {
    cancelAnimationFrame(raf);
    mq?.removeEventListener?.("change", setReduceMotion);

    if (timer) window.clearInterval(timer);
    window.removeEventListener("scroll", onScroll);

    io?.disconnect?.();
    cleanupThree?.();
});
</script>
